15def same(file_a, file_b, verbose=0): 

16 """ Determines if two images are the same or different. 

17 

18 Creates to image files from the image string arguments, then gets the 

19 pixel data from each image. Subtracts pixel information of file_b from 

20 file_a, and creates a numpy array, which is the pixel-by-pixel difference 

21 of the two images. 

22 

23 If the L2 norm of the difference array is less than a small tolerance, the 

24 two images are classified as the same. Otherwise, they are classified as 

25 different. 

26 

27 Args: 

28 file_a (string) The first of two image file names. 

29 file_b (string): The second of two image file names. 

30 verbose: (Boolean): True provides more command line feedback that False. 

31 

32 Returns: 

33 Boolean: True for images are the same; False for images are different. 

34 """ 

35 

36 same_xy_dimension = False 

37 same_mode = False 

38 same_channels = False 

39 same_pixels = False 

40 

41 if verbose: # pragma: no cover 

42 print(f'Image 1 file name: {file_a}') 

43 print(f'Image 2 file name: {file_b}') 

44 

45 try: 

46 with Image.open(file_a) as im_a, Image.open(file_b) as im_b: 

47 

48 im_a_size = im_a.size 

49 im_b_size = im_b.size 

50 

51 same_xy_dimension = im_a_size == im_b_size 

52 

53 if verbose: # pragma: no cover 

54 print('Test 1 of 2: Dimensionality comparison') 

55 print(f' (x,y) dimension of image 1 is {im_a_size}') 

56 print(f' (x,y) dimension of image 2 is {im_b_size}') 

57 

58 same_mode = im_a.mode == im_b.mode 

59 

60 if same_xy_dimension and same_mode: 

61 

62 data_a = np.array(im_a.getdata()).flatten() 

63 data_b = np.array(im_b.getdata()).flatten() 

64 

65 same_channels = data_a.size == data_b.size 

66 

67 if same_channels: 

68 data_diff = data_a - data_b 

69 

70 data_diff_norm = np.linalg.norm(data_diff) 

71 image_tol = 10.0 # tolerance for the L2norm to be same or different 

72 

73 same_pixels = np.abs(data_diff_norm) < image_tol 

74 

75 if verbose: # pragma: no cover 

76 print(f' Images (x,y) dimensions are the same? [T/F]: {same_xy_dimension}') 

77 print(f' Images have same mode? [T/F]: {same_mode}') 

78 print(f' Images have same number of channels? [T/F]: {same_channels}') 

79 print('Test 2 of 2: Pixel-by-pixel comparison') 

80 print(f' Size of data image 1 is {data_a.size}') 

81 print(f' Size of data image 2 is {data_b.size}') 

82 print(f' Pixels are the same? [T/F]: {same_pixels}') 

83 

84 else: # pragma: no cover 

85 if verbose: 

86 print(f' Images (x,y) dimensions are the same? [T/F]: {same_xy_dimension}') 

87 print(f' Images have same mode? [T/F]: {same_mode}') 

88 print(f' Images have same number of channels? [T/F]: {same_channels}') 

89 print('Images are different.') 

90 

91 if verbose: # pragma: no cover 

92 print(f'Returning {same_pixels}') 

93 

94 return same_pixels 

95 

96 except IOError: 

97 print('Error: could not open files:') 

98 print(f' {file_a}') 

99 print(f' {file_b}')
